% If images/filename.png exists it will be included, otherwise shows a placeholder box.
\newcommand{\figplaceholder}[2]{%
  \IfFileExists{images/#2.png}{%
    \begin{figure}[H]
      \centering
      \includegraphics[width=0.92\textwidth]{images/#2.png}
      \caption{#1}
      \label{fig:#2}
    \end{figure}
  }{%
    \IfFileExists{images/#2.jpg}{%
      \begin{figure}[H]
        \centering
        \includegraphics[width=0.92\textwidth]{images/#2.jpg}
        \caption{#1}
        \label{fig:#2}
      \end{figure}
    }{%
      \begin{figure}[H]
        \centering
        \fbox{\parbox{0.85\textwidth}{\centering\vspace{1.5cm}%
          \textbf{[ Screenshot needed: #2.png ]}\\[0.3cm]%
          #1\vspace{1.5cm}}}
        \caption{#1}
        \label{fig:#2}
      \end{figure}
    }%
  }%
}
